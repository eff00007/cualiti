168 void Fecha::comprobarFecha(unsigned aDia, unsigned aMes, unsigned aAnio, unsigned aHora, unsigned aMin) const
169 {
170  if (aMin > 59 || aHora > 23)
171  throw ErrorFechaIncorrecta();
172  if (aMes < 1 || aMes > 12)
173  throw ErrorFechaIncorrecta();
174  if (aDia < 1 || aDia > diasMes[aMes - 1])
175  throw ErrorFechaIncorrecta();
176  if (aDia == 29 && aMes == 2 && (aAnio % 4 != 0 || (aAnio % 100 == 0 && aAnio % 400 != 0)))
177  throw ErrorFechaIncorrecta();
178 }
